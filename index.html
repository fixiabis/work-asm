ORG     0000H

INIT:
    MOV     DPTR,#NUMBER_APPEARANCE_TABLE
    ACALL   MAIN
    AJMP    FINISH

MAIN:
    MOV     R0,#10

    MAIN_LOOP1:
        MOV     R1,#10

        MAIN_LOOP2:
            ACALL   SHOW_NUMBER
            DJNZ    R1,MAIN_LOOP2

        DJNZ    R0,MAIN_LOOP1

    RET

SHOW_NUMBER:
    MOV     R2,#10

    SHOW_NUMBER_CYCLE: ; 維持顯示, 讓倒數速度不會那麼快
        MOV     P1,#11110011B
        MOV     P0,#0C0H
        ACALL   DELAY

        MOV     P1,#11111101B
        MOV     A,R0
        MOVC    A,@A+DPTR
        MOV     P0,A
        ACALL   DELAY

        MOV     P1,#11111110B
        MOV     A,R1
        MOVC    A,@A+DPTR
        MOV     P0,A
        ACALL   DELAY

        DJNZ    R2,SHOW_NUMBER_CYCLE
    RET

DELAY:
    MOV     R6,#10
    DELAY_LOOP1:
        MOV     R7,#200
        DELAY_LOOP2:
            DJNZ    R7,DELAY_LOOP2
        DJNZ    R6,DELAY_LOOP1
    RET

NUMBER_APPEARANCE_TABLE:
    DB      0C0H ; 可忽略此項, 資料從1開始取得
    DB      0C0H ; 0
    DB      0F9H ; 1
    DB      0A4H ; 2
    DB      0B0H ; 3
    DB      099H ; 4
    DB      092H ; 5
    DB      082H ; 6
    DB      0F8H ; 7
    DB      080H ; 8
    DB      090H ; 9

SOUND:
    CLR     P2.6
    ACALL   VOICE
    
    SETB     P2.6
    ACALL   VOICE

    DJNZ    R5,SOUND
    RET

VOICE:
    VOICE_LOOP1:
        MOV     R7,#06
        DJNZ    R7,$
        DJNZ    R6,VOICE_LOOP1
    RET

FINISH:
    MOV     R6,#126
    MOV     R5,#193
    ACALL   SOUND
    
    MOV     R6,#95
    MOV     R5,#255
    ACALL   SOUND

    MOV     P1,#11110000B
    MOV     P0,#0C0H

    FIN:    AJMP    FIN

    END
